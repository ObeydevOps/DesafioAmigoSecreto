let nome = "";
let nomesSorteados = []; // Lista para rastrear nomes já sorteados

// Função para verificar se o nome já existe na lista (case-insensitive)
function nomeJaExiste(novoNome) {
    return listaDeAmigos.some(existingNome => existingNome.toLowerCase() === novoNome.toLowerCase());
}

// Função para salvar nome
function salvarNome() {
    nome = document.getElementById("nome").value.trim();
    if (nome === "") {
        alert("Por favor, insira seu nome!");
        return;
    }
    if (nomeJaExiste(nome)) {
        alert("Este nome já está na lista!");
        return;
    }
    listaDeAmigos.push(nome);
    document.getElementById("nome").value = "";
    atualizarLista();
}

// Criando uma lista de amigos vazia
let listaDeAmigos = [];

// Função para adicionar um amigo na lista
function adicionarAmigo() {
    let nomeAmigo = document.getElementById("amigo").value.trim();
    if (nomeAmigo === "") {
        alert("Por favor, insira um nome!");
        return;
    }
    if (nomeJaExiste(nomeAmigo)) {
        alert("Este nome já está na lista!");
        return;
    }
    listaDeAmigos.push(nomeAmigo);
    document.getElementById("amigo").value = "";
    atualizarLista();
}

// Função para atualizar a lista de amigos
function atualizarLista() {
    let listaAtualizada = document.getElementById("listaAmigos");
    listaAtualizada.innerHTML = "";
    for (let i = 0; i < listaDeAmigos.length; i++) {
        let itemLista = document.createElement("li");
        itemLista.innerHTML = listaDeAmigos[i];
        listaAtualizada.appendChild(itemLista);
    }
}

// Função para sortear amigo, evitando repetição de nomes sorteados e o próprio nome
function sortearAmigo() {
    if (listaDeAmigos.length === 0) {
        alert("Por favor, insira um nome!");
        return;
    }

    // Filtra nomes disponíveis (não sorteados e diferentes do próprio nome)
    let nomesDisponiveis = listaDeAmigos.filter(
        n => !nomesSorteados.includes(n) && n.toLowerCase() !== nome.toLowerCase()
    );

    if (nomesDisponiveis.length === 0) {
        alert("Todos os nomes disponíveis já foram sorteados! Reinicie o sorteio.");
        return;
    }

    // Sorteia um nome dentre os disponíveis
    let indiceAleatorio = Math.floor(Math.random() * nomesDisponiveis.length);
    let nomeSorteado = nomesDisponiveis[indiceAleatorio];

    // Adiciona o nome sorteado à lista de sorteados
    nomesSorteados.push(nomeSorteado);

    document.getElementById("resultado").innerHTML = `Seu amigo secreto é: ${nomeSorteado}`;
}

// Função para reiniciar o sorteio
function reiniciarSorteio() {
    listaDeAmigos = [];
    nomesSorteados = []; // Limpa a lista de nomes sorteados
    document.getElementById("resultado").innerHTML = "";
    atualizarLista();
}
